import pygame import sys # Initialize Pygame pygame.init() # Set up the screen SCREEN_WIDTH = 800 SCREEN_HEIGHT = 600 screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) pygame.display.set_caption("Doraemon Characters") # Define colors WHITE = (255, 255, 255) BLUE = (65, 105, 225) BLACK = (0, 0, 0) RED = (255, 0, 0) YELLOW = (255, 255, 0) GREEN = (0, 255, 0) # Main drawing function def draw_doraemon(): # Draw Doraemon's body pygame.draw.circle(screen, BLUE, (400, 300), 150) # Draw Doraemon's face pygame.draw.circle(screen, WHITE, (400, 250), 120) # Draw Doraemon's eyes pygame.draw.circle(screen, WHITE, (350, 200), 30) pygame.draw.circle(screen, WHITE, (450, 200), 30) pygame.draw.circle(screen, BLACK, (350, 200), 15) pygame.draw.circle(screen, BLACK, (450, 200), 15) # Draw Doraemon's nose pygame.draw.circle(screen, RED, (400, 250), 10) # Draw Doraemon's mouth pygame.draw.arc(screen, BLACK, (350, 230, 100, 70), 0, -3.14, 2) # Draw Nobita pygame.draw.rect(screen, GREEN, (100, 300, 100, 200)) # Draw Gian pygame.draw.rect(screen, YELLOW, (250, 300, 100, 200)) # Draw Suneo pygame.draw.rect(screen, YELLOW, (550, 300, 100, 200)) # Draw Shizuka pygame.draw.circle(screen, WHITE, (700, 400), 50) # Main game loop running = True while running: # Event handling for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # Clear the screen screen.fill(WHITE) # Draw characters draw_doraemon() # Update the display pygame.display.flip() # Cap the frame rate pygame.time.Clock().tick(60) # Quit Pygame pygame.quit() sys.exit()
